# Bell State Creation - Quantum Entanglement from qiskit import QuantumCircuit, execute, Aer from qiskit.visualization import plot_histogram import matplotlib.pyplot as plt # Create a quantum circuit with 2 qubits and 2 classical bits qc = QuantumCircuit(2, 2) # Apply Hadamard gate to qubit 0 qc.h(0) # Apply CNOT gate with qubit 0 as control, qubit 1 as target qc.cx(0, 1) # Measure both qubits qc.measure_all() # Execute on quantum simulator backend = Aer.get_backend('qasm_simulator') job = execute(qc, backend, shots=1024) result = job.result() counts = result.get_counts(qc) print("Bell State Results:", counts)
# Quantum Superposition Demo from qiskit import QuantumCircuit, ClassicalRegister, QuantumRegister # Create quantum and classical registers qreg = QuantumRegister(3, 'q') creg = ClassicalRegister(3, 'c') circuit = QuantumCircuit(qreg, creg) # Create superposition on all qubits for i in range(3): circuit.h(qreg[i]) # Add rotation gates for phase manipulation circuit.rz(0.5, qreg[0]) circuit.ry(0.3, qreg[1]) circuit.rx(0.7, qreg[2]) # Measure all qubits circuit.measure(qreg, creg) print("Quantum Circuit Depth:", circuit.depth()) print("Number of Operations:", len(circuit.data))
# Quantum Fourier Transform (QFT) import numpy as np from qiskit import QuantumCircuit from qiskit.circuit.library import QFT def create_qft_circuit(n_qubits): """Create a Quantum Fourier Transform circuit""" qc = QuantumCircuit(n_qubits) # Add some initial state preparation for i in range(n_qubits): qc.h(i) qc.rz(np.pi/4 * i, i) # Apply QFT qft = QFT(n_qubits, inverse=False) qc.compose(qft, inplace=True) return qc # Create 4-qubit QFT circuit qft_circuit = create_qft_circuit(4) print(f"QFT Circuit created with {qft_circuit.num_qubits} qubits")

The first script creates a Bell state, demonstrating quantum entanglement. When measured, the two qubits will always show correlated results - either both 0 or both 1, with equal probability.

Key Gates: Hadamard (H) creates superposition, CNOT creates entanglement.

The superposition demo shows how qubits can exist in multiple states simultaneously. The Hadamard gates create equal superposition, while rotation gates (RX, RY, RZ) manipulate quantum phases.

Applications: Quantum algorithms, parallel computation, quantum machine learning.

Complexity: Exponentially faster than classical FFT for certain problems.
